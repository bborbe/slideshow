Images mit Docker

Dockerfile

FROM ubuntu:14.04
RUN apt-get install --quiet --yes openssh-server
EXPOSE 22
CMD ["/usr/sbin/sshd", "-D"]

Build

docker build --rm=true -t bborbe/ssh .

Upload

docker push bborbe/ssh

Pod

apiVersion: v1
kind: Pod
metadata:
  name: static-go-webserver
spec:
  containers:
  - image: bborbe/static-go-webserver:latest
    name: static-go-webserver
    ports:
    - containerPort: 80
      name: static-go-web

Replication Controller

apiVersion: v1
kind: ReplicationController
metadata:
  name: static-go-webserver
spec:
  replicas: 2
  selector:
    app: static-go-webserver
  template:
    metadata:
      labels:
        app: static-go-webserver
    spec:
      containers:
      - image: bborbe/static-go-webserver:latest
        name: static-go-webserver
        ports:
        - containerPort: 80
          name: static-go-web

Service

apiVersion: v1
kind: Service
metadata:
  name: static-go-webserver
spec:
  ports:
  - nodePort: 30080
    port: 8080
  selector:
    app: static-go-webserver
  type: NodePort
