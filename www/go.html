Currency mit Go


	package main

	import "fmt"



func main() {
	go printHello()
	fmt.Printf("World")
}

func printHello() {
	fmt.Printf("Hello ")
}

Wo liegt das Problem?

Channels


	package main

	import "fmt"



func main() {
	done := make(chan bool, 1)
	go printHello(done)
	<- done
	fmt.Printf("World")
}

func printHello(done chan bool) {
	fmt.Printf("Hello ")
	done <- true
}

defer


	package main

	import "fmt"



func main() {
	done := make(chan bool, 1)
	go printHello(done)
	<- done
}

func printHello(done chan bool) {
	defer func() {
		fmt.Printf("World")
		done <- true
	}()
	fmt.Printf("Hello ")
}
